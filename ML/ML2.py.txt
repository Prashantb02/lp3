# Simple exam-ready spam classification (KNN & SVM)
import pandas as pd
from sklearn.model_selection import train_test_split
from sklearn.neighbors import KNeighborsClassifier
from sklearn.svm import SVC
from sklearn.preprocessing import StandardScaler
from sklearn.metrics import classification_report, confusion_matrix, accuracy_score
from sklearn.utils import resample

# 1. Read & basic preprocess
df = pd.read_csv("emails.csv")          # path to dataset
df = df.drop("Email No.", axis=1)       # drop index column if present

# If Prediction is a DataFrame column, use Series:
X = df.drop("Prediction", axis=1)
y = df["Prediction"]

# 2. Optional: upsample minority class (if needed)
spam = df[df["Prediction"] == 1]
ham  = df[df["Prediction"] == 0]
spam_upsampled = resample(spam, replace=True, n_samples=len(ham), random_state=42)
df_bal = pd.concat([ham, spam_upsampled]).sample(frac=1, random_state=42)
X = df_bal.drop("Prediction", axis=1)
y = df_bal["Prediction"]

# 3. Train-test split (use stratify for class balance)
X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.2,
                                                    stratify=y, random_state=42)

# 4. Feature scaling (important for KNN & SVM)
scaler = StandardScaler()
X_train = scaler.fit_transform(X_train)
X_test  = scaler.transform(X_test)

# 5. KNN (choose k by elbow outside exam or just use k=5)
knn = KNeighborsClassifier(n_neighbors=5)
knn.fit(X_train, y_train)
y_knn = knn.predict(X_test)

print("KNN Results:")
print(classification_report(y_test, y_knn))
print("Confusion matrix:\n", confusion_matrix(y_test, y_knn))
print("Accuracy:", accuracy_score(y_test, y_knn))
print("\n" + "="*60 + "\n")

# 6. SVM (RBF kernel default)
svm = SVC(kernel='rbf', C=1.0, random_state=42)
svm.fit(X_train, y_train)
y_svm = svm.predict(X_test)

print("SVM Results:")
print(classification_report(y_test, y_svm))
print("Confusion matrix:\n", confusion_matrix(y_test, y_svm))
print("Accuracy:", accuracy_score(y_test, y_svm))
